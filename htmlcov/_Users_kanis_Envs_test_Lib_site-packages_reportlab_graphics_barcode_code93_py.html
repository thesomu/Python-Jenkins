87def _encode93(str): 

88 s = list(str) 

89 s.reverse() 

90 

91 # compute 'C' checksum 

92 i = 0; v = 1; c = 0 

93 while i < len(s): 

94 c = c + v * _patterns[s[i]][1] 

95 i = i + 1; v = v + 1 

96 if v > 20: 

97 v = 1 

98 s.insert(0, _charsbyval[c % 47]) 

99 

100 # compute 'K' checksum 

101 i = 0; v = 1; c = 0 

102 while i < len(s): 

103 c = c + v * _patterns[s[i]][1] 

104 i = i + 1; v = v + 1 

105 if v > 15: 

106 v = 1 

107 s.insert(0, _charsbyval[c % 47]) 

108 

109 s.reverse() 

110 

111 return ''.join(s)

144class Standard93(_Code93Base): 

145 """ 

146 Code 93 is a Uppercase alphanumeric symbology with some punctuation. 

147 See Extended Code 93 for a variant that can represent the entire 

148 128 characrter ASCII set. 

149  

150 Options that may be passed to constructor: 

151 

152 value (int, or numeric string. required.): 

153 The value to encode. 

154  

155 barWidth (float, default .0075): 

156 X-Dimension, or width of the smallest element 

157 Minumum is .0075 inch (7.5 mils). 

158  

159 barHeight (float, see default below): 

160 Height of the symbol. Default is the height of the two 

161 bearer bars (if they exist) plus the greater of .25 inch 

162 or .15 times the symbol's length. 

163 

164 quiet (bool, default 1): 

165 Wether to include quiet zones in the symbol. 

166  

167 lquiet (float, see default below): 

168 Quiet zone size to left of code, if quiet is true. 

169 Default is the greater of .25 inch, or 10 barWidth 

170  

171 rquiet (float, defaults as above): 

172 Quiet zone size to right left of code, if quiet is true. 

173 

174 stop (bool, default 1): 

175 Whether to include start/stop symbols. 

176 

177 Sources of Information on Code 93: 

178 

179 http://www.semiconductor.agilent.com/barcode/sg/Misc/code_93.html 

180 

181 Official Spec, "NSI/AIM BC5-1995, USS" is available for US$45 from 

182 http://www.aimglobal.org/aimstore/ 

183 """ 

184 def validate(self): 

185 vval = "" 

186 self.valid = 1 

187 for c in self.value.upper(): 

188 if c not in _patterns: 

189 self.valid = 0 

190 continue 

191 vval = vval + c 

192 self.validated = vval 

193 return vval

195 def encode(self): 

196 self.encoded = _encode93(self.validated)

200class Extended93(_Code93Base): 

201 """ 

202 Extended Code 93 is a convention for encoding the entire 128 character 

203 set using pairs of characters to represent the characters missing in 

204 Standard Code 93. It is very much like Extended Code 39 in that way. 

205  

206 See Standard93 for arguments. 

207 """ 

208 

209 def validate(self): 

210 vval = [] 

211 self.valid = 1 

212 a = vval.append 

213 for c in self.value: 

214 if c not in _patterns and c not in _extended: 

215 self.valid = 0 

216 continue 

217 a(c) 

218 self.validated = ''.join(vval) 

219 return self.validated 

220 

221 def encode(self): 

222 self.encoded = "" 

223 for c in self.validated: 

224 if c in _patterns: 

225 self.encoded = self.encoded + c 

226 elif c in _extended: 

227 self.encoded = self.encoded + _extended[c] 

228 else: 

229 raise ValueError 

230 self.encoded = _encode93(self.encoded) 

231 return self.encoded
