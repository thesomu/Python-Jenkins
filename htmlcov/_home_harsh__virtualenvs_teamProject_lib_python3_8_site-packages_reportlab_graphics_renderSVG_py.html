70def transformNode(doc, newTag, node=None, **attrDict): 

71 """Transform a DOM node into new node and copy selected attributes. 

72 

73 Creates a new DOM node with tag name 'newTag' for document 'doc' 

74 and copies selected attributes from an existing 'node' as provided 

75 in 'attrDict'. The source 'node' can be None. Attribute values will 

76 be converted to strings. 

77 

78 E.g. 

79 

80 n = transformNode(doc, "node1", x="0", y="1") 

81 -> DOM node for <node1 x="0" y="1"/> 

82 

83 n = transformNode(doc, "node1", x=0, y=1+1) 

84 -> DOM node for <node1 x="0" y="2"/> 

85 

86 n = transformNode(doc, "node1", node0, x="x0", y="x0", zoo=bar()) 

87 -> DOM node for <node1 x="[node0.x0]" y="[node0.y0]" zoo="[bar()]"/> 

88 """ 

89 

90 newNode = doc.createElement(newTag) 

91 for newAttr, attr in attrDict.items(): 

92 sattr = str(attr) 

93 if not node: 

94 newNode.setAttribute(newAttr, sattr) 

95 else: 

96 attrVal = node.getAttribute(sattr) 

97 newNode.setAttribute(newAttr, attrVal or sattr) 

98 

99 return newNode

101class EncodedWriter(list): 

102 ''' 

103 EncodedWriter(encoding) assumes .write will be called with 

104 either unicode or utf8 encoded bytes. it will accumulate 

105 unicode 

106 ''' 

107 BOMS = { 

108 'utf-32':codecs.BOM_UTF32, 

109 'utf-32-be':codecs.BOM_UTF32_BE, 

110 'utf-32-le':codecs.BOM_UTF32_LE, 

111 'utf-16':codecs.BOM_UTF16, 

112 'utf-16-be':codecs.BOM_UTF16_BE, 

113 'utf-16-le':codecs.BOM_UTF16_LE, 

114 } 

115 def __init__(self,encoding,bom=False): 

116 list.__init__(self) 

117 self.encoding = encoding = codecs.lookup(encoding).name 

118 if bom and '16' in encoding or '32' in encoding: 

119 self.write(self.BOMS[encoding])

121 def write(self,u): 

122 if isBytes(u): 

123 try: 

124 u = u.decode('utf-8') 

125 except: 

126 et, ev, tb = sys.exc_info() 

127 ev = str(ev) 

128 del et, tb 

129 raise ValueError("String %r not encoded as 'utf-8'\nerror=%s" % (u,ev)) 

130 elif not isUnicode(u): 

131 raise ValueError("EncodedWriter.write(%s) argument should be 'utf-8' bytes or str" % ascii(u)) 

132 self.append(u)

587 def drawArc(self, x1,y1, x2,y2, startAng=0, extent=360, fromcenter=0): 

588 """Draw a partial ellipse inscribed within the rectangle x1,y1,x2,y2. 

589 

590 Starting at startAng degrees and covering extent degrees. Angles 

591 start with 0 to the right (+x) and increase counter-clockwise. 

592 These should have x1<x2 and y1<y2. 

593 """ 

594 

595 cx, cy = (x1+x2)/2.0, (y1+y2)/2.0 

596 rx, ry = (x2-x1)/2.0, (y2-y1)/2.0 

597 mx = rx * cos(startAng*pi/180) + cx 

598 my = ry * sin(startAng*pi/180) + cy 

599 ax = rx * cos((startAng+extent)*pi/180) + cx 

600 ay = ry * sin((startAng+extent)*pi/180) + cy 

601 

602 cfp_str = self.cfp_str 

603 s = [].append 

604 if fromcenter: 

605 s("M %s L %s" % (cfp_str(cx, cy), cfp_str(ax, ay))) 

606 

607 if fromcenter: 

608 s("A %s %d %d %d %s" % \ 

609 (cfp_str(rx, ry), 0, extent>=180, 0, cfp_str(mx, my))) 

610 else: 

611 s("M %s A %s %d %d %d %s Z" % \ 

612 (cfp_str(mx, my), cfp_str(rx, ry), 0, extent>=180, 0, cfp_str(mx, my))) 

613 

614 if fromcenter: 

615 s("L %s Z" % cfp_str(cx, cy)) 

616 

617 path = transformNode(self.doc, "path", 

618 d=' '.join(s.__self__), style=self._formatStyle()) 

619 self.currGroup.appendChild(path)

715 """This is the recursive method called for each node in the tree. 

716 """

721 style = self._canvas.style.copy() 

722 if not (isinstance(node, Path) and node.isClipPath): 

723 pass # self._canvas.saveState() 

724

726 deltas = getStateDelta(node) 

727 self._tracker.push(deltas)

750 def _get_link_info_dict(self, obj): 

751 #We do not want None or False as the link, even if it is the 

752 #attribute's value - use the empty string instead. 

753 url = getattr(obj, "hrefURL", "") or "" 

754 title = getattr(obj, "hrefTitle", "") or "" 

755 if url : 

756 #Is it valid to have a link with no href? The XML requires 

757 #the xlink:href to be present, but you might just want a 

758 #tool tip shown (via the xlink:title attribute). Note that 

759 #giving an href of "" is equivalent to "the current page" 

760 #(a relative link saying go nowhere). 

761 return {"xlink:href":url, "xlink:title":title, "target":"_top"} 

762 #Currently of all the mainstream browsers I have tested, only Safari/webkit 

763 #will show SVG images embedded in HTML using a simple <img src="..." /> tag. 

764 #However, the links don't work (Safari 3.2.1 on the Mac). 

765 # 

766 #Therefore I use the following, which also works for Firefox, Opera, and 

767 #IE 6.0 with Adobe SVG Viewer 6 beta: 

768 #<object data="..." type="image/svg+xml" width="430" height="150" class="img"> 

769 # 

770 #Once displayed, Firefox and Safari treat the SVG like a frame, and 

771 #by default clicking on links acts "in frame" and replaces the image. 

772 #Opera does what I expect, and replaces the whole page with the link. 

773 # 

774 #Therefore I use target="_top" to force the links to replace the whole page. 

775 #This now works as expected on Safari 3.2.1, Firefox 3.0.6, Opera 9.20.  

776 #Perhaps the target attribute should be an option, perhaps defaulting to 

777 #"_top" as used here? 

778 else : 

779 return None
