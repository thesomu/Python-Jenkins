55 default=False, 

56 action='store_true', 

57 help="Output to stderr steps taken with the algorithm")

59 parser.add_option('-b', '--base-dir',

61 default=None, 

62 type='string', 

63 help="Override base direction [L|R]") 

64 

65 options, rest = parser.parse_args() 

66 

67 if options.base_dir and options.base_dir not in 'LR': 

68 parser.error('option -b can be L or R') 

69 

70 # allow unicode in sys.stdout.write 

71 if six.PY2: 

72 sys.stdout = codecs.getwriter(locale.getpreferredencoding())(sys.stdout) 

73 

74 if rest: 

75 lines = rest 

76 else: 

77 lines = sys.stdin 

78 

79 for line in lines: 

80 display = get_display(line, options.encoding, options.upper_is_rtl, 

81 options.base_dir, options.debug) 

82 # adjust the encoding as unicode, to match the output encoding 

83 if not isinstance(display, six.text_type): 

84 display = display.decode(options.encoding)
