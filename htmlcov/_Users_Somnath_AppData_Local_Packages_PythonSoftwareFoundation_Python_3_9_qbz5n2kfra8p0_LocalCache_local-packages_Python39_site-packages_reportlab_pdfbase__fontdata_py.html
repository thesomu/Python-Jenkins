136 def __setitem__(self,x,v): 

137 y = x.lower() 

138 if y[-8:]=='encoding': y = y[:-8] 

139 y = self._XMap[y] 

140 if y in self: raise IndexError('Encoding %s is already set' % y) 

141 dict.__setitem__(self,y,v)

143 def __getitem__(self,x): 

144 y = x.lower() 

145 if y[-8:]=='encoding': y = y[:-8]
