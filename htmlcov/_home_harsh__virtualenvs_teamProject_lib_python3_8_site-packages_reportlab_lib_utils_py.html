570def getArgvDict(**kw): 

571 ''' Builds a dictionary from its keyword arguments with overrides from sys.argv. 

572 Attempts to be smart about conversions, but the value can be an instance 

573 of ArgDictValue to allow specifying a conversion function. 

574 ''' 

575 def handleValue(v,av,func): 

576 if func: 

577 v = func(av) 

578 else: 

579 if isStr(v): 

580 v = av 

581 elif isinstance(v,float): 

582 v = float(av) 

583 elif isinstance(v,int): 

584 v = int(av) 

585 elif isinstance(v,list): 

586 v = list(literal_eval(av),{}) 

587 elif isinstance(v,tuple): 

588 v = tuple(literal_eval(av),{}) 

589 else: 

590 raise TypeError("Can't convert string %r to %s" % (av,type(v))) 

591 return v 

592 

593 A = sys.argv[1:] 

594 R = {} 

595 for k, v in kw.items(): 

596 if isinstance(v,ArgvDictValue): 

597 v, func = v.value, v.func 

598 else: 

599 func = None 

600 handled = 0 

601 ke = k+'=' 

602 for a in A: 

603 if a.startswith(ke): 

604 av = a[len(ke):] 

605 A.remove(a) 

606 R[k] = handleValue(v,av,func) 

607 handled = 1 

608 break 

609 

610 if not handled: R[k] = handleValue(v,v,func) 

611 

612 return R

1382def commasplit(s): 

1383 ''' 

1384 Splits the string s at every unescaped comma and returns the result as a list. 

1385 To escape a comma, double it. Individual items are stripped. 

1386 To avoid the ambiguity of 3 successive commas to denote a comma at the beginning 

1387 or end of an item, add a space between the item seperator and the escaped comma. 

1388  

1389 >>> commasplit(u'a,b,c') == [u'a', u'b', u'c'] 

1390 True 

1391 >>> commasplit('a,, , b , c ') == [u'a,', u'b', u'c'] 

1392 True 

1393 >>> commasplit(u'a, ,,b, c') == [u'a', u',b', u'c'] 

1394 ''' 

1395 if isBytes(s): s = s.decode('utf8') 

1396 n = len(s)-1 

1397 s += u' ' 

1398 i = 0 

1399 r=[u''] 

1400 while i<=n: 

1401 if s[i]==u',': 

1402 if s[i+1]==u',': 

1403 r[-1]+=u',' 

1404 i += 1 

1405 else: 

1406 r[-1] = r[-1].strip() 

1407 if i!=n: r.append(u'') 

1408 else: 

1409 r[-1] += s[i] 

1410 i+=1 

1411 r[-1] = r[-1].strip() 

1412 return r
