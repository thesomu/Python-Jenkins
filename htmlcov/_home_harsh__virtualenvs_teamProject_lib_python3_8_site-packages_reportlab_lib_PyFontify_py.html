40def replace(src, sep, rep): 

41 return rep.join(src.split(sep))

43# This list of keywords is taken from ref/node13.html of the 

44# Python 1.3 HTML documentation. ("access" is intentionally omitted.) 

45keywordsList = [

55# Build up a regular expression which will match anything 

56# interesting, including multi-line triple-quoted strings. 

57commentPat = r"#[^\n]*" 

58 

59pat = r"q[^\\q\n]*(\\[\000-\377][^\\q\n]*)*q" 

60quotePat = replace(pat, "q", "'") + "|" + replace(pat, 'q', '"') 

61 

62# Way to go, Tim! 

63pat = r""" 

64 qqq 

65 [^\\q]* 

66 ( 

67 ( \\[\000-\377] 

68 | q 

69 ( \\[\000-\377] 

70 | [^\q] 

71 | q 

72 ( \\[\000-\377] 

73 | [^\\q] 

74 ) 

75 ) 

76 ) 

77 [^\\q]* 

78 )* 

79 qqq 

80""" 

81pat = ''.join(pat.split()) # get rid of whitespace 

82tripleQuotePat = replace(pat, "q", "'") + "|" + replace(pat, 'q', '"') 

83 

84# Build up a regular expression which matches all and only 

85# Python keywords. This will let us skip the uninteresting 

86# identifier references. 

87# nonKeyPat identifies characters which may legally precede 

88# a keyword pattern. 

89nonKeyPat = r"(^|[^a-zA-Z0-9_.\"'])" 

90 

91keyPat = nonKeyPat + "(" + "|".join(keywordsList) + ")" + nonKeyPat 

92 

93matchPat = commentPat + "|" + keyPat + "|" + tripleQuotePat + "|" + quotePat 

94matchRE = re.compile(matchPat) 

95 

96idKeyPat = "[ \t]*[A-Za-z_][A-Za-z_0-9.]*" # Ident w. leading whitespace. 

97idRE = re.compile(idKeyPat)

100def fontify(pytext, searchfrom = 0, searchto = None): 

101 if searchto is None: 

102 searchto = len(pytext) 

103 # Cache a few attributes for quicker reference. 

104 search = matchRE.search 

105 idSearch = idRE.search 

106 

107 tags = [] 

108 tags_append = tags.append 

109 commentTag = 'comment' 

110 stringTag = 'string' 

111 keywordTag = 'keyword' 

112 identifierTag = 'identifier' 

113 

114 start = 0 

115 end = searchfrom 

116 while 1: 

117 m = search(pytext, end) 

118 if m is None: 

119 break # EXIT LOOP 

120 start = m.start() 

121 if start >= searchto: 

122 break # EXIT LOOP 

123 match = m.group(0) 

124 end = start + len(match) 

125 c = match[0] 

126 if c not in "#'\"": 

127 # Must have matched a keyword. 

128 if start != searchfrom: 

129 # there's still a redundant char before and after it, strip! 

130 match = match[1:-1] 

131 start = start + 1 

132 else: 

133 # this is the first keyword in the text. 

134 # Only a space at the end. 

135 match = match[:-1] 

136 end = end - 1 

137 tags_append((keywordTag, start, end, None)) 

138 # If this was a defining keyword, look ahead to the 

139 # following identifier. 

140 if match in ["def", "class"]: 

141 m = idSearch(pytext, end) 

142 if m is not None: 

143 start = m.start() 

144 if start == end: 

145 match = m.group(0) 

146 end = start + len(match) 

147 tags_append((identifierTag, start, end, None)) 

148 elif c == "#": 

149 tags_append((commentTag, start, end, None)) 

150 else: 

151 tags_append((stringTag, start, end, None)) 

152 return tags
